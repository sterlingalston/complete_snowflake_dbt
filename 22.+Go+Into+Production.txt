      database: BTC
      password: YOUR_SNOWFLAKE_PASSWORD
      role: ACCOUNTADMIN
      schema: YOUR_SNOWFLAKE_SCHEMA
      threads: 1
      type: snowflake
      user: YOUR_SNOWFLAKE_USER
      warehouse: LARGE_WH

    prod:
      account: YOUR_SNOWFLAKE_ACCOUNT
      database: BTC
      password: YOUR_SNOWFLAKE_PASSWORD
      role: ACCOUNTADMIN
      schema: PROD
      threads: 1
      type: snowflake
      user: YOUR_SNOWFLAKE_USER
      warehouse: LARGE_WH

  target: dev
