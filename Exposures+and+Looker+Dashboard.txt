***Example: schema.yml with an Exposure***

exposures:

  - name: btc_whale_alert_exposure
    label: BTC Whale alert
    type: dashboard
    maturity: high
    url: https://lookerstudio.google.com/reporting/replace_with_your_own_looker_report
    description: >
      BTC Whale alert transactions report"?

    depends_on:
      - ref('whale_alert')

    owner:
      name: Your Own Name
